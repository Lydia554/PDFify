üì¶ Packing Slip

JavaScript Example

fetch("https://pdf-api.portfolio.lidija-jokic.com/api/generate-packing-slip", {
  method: "POST",
  headers: {
    "Content-Type": "application/json",
    "Authorization": "Bearer YOUR_API_KEY"
  },
  body: JSON.stringify({
    data: {
      orderId: "98765",
      customerName: "John Smith",
      shippingAddress: "1234 Maple Street, Springfield, USA",
      date: "2025-05-14",
      items: [
        { name: "Yoga Mat", sku: "YM-001", quantity: 1 },
        { name: "Dumbbells Set", sku: "DB-002", quantity: 2 },
        { name: "Water Bottle", sku: "WB-003", quantity: 3 }
      ]
    }
  })
})
.then(response => response.blob())
.then(blob => {
  const url = window.URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = "PackingSlip_98765.pdf";
  document.body.appendChild(a);
  a.click();
  a.remove();
  window.URL.revokeObjectURL(url);
})
.catch(console.error);

curl Example

curl -X POST https://pdf-api.portfolio.lidija-jokic.com/api/generate-packing-slip \
-H "Authorization: Bearer <your-token>" \
-H "Content-Type: application/json" \
-d '{
  "title": "Order #1234",
  "customer": "Jane Doe",
  "items": [
    {"name": "Shirt", "quantity": 2, "price": 24.99}
  ]
}' --output packing-slip.pdf
